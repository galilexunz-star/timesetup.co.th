    function sortTableByTime(table, timeColIndex) {
        if (!table || !table.tBodies || !table.tBodies[0]) return;
        const tbody = table.tBodies[0];
        const rows = Array.from(tbody.querySelectorAll('tr'));

        const getTimeValue = (row) => {
            const cell = row.cells[timeColIndex];
            if (!cell) return Number.MAX_SAFE_INTEGER;

            const text = (cell.textContent || '').trim();
            if (!text || text === '-' || text.includes('ไม่พบข้อมูล')) {
                return Number.MAX_SAFE_INTEGER;
            }
            const parts = text.split(':').map(Number);
            if (parts.length >= 2 && !parts.some(isNaN)) {
                return parts[0] * 60 + parts[1];
            }
            return Number.MAX_SAFE_INTEGER;
        };

        rows.sort((a, b) => getTimeValue(a) - getTimeValue(b));
        rows.forEach(row => tbody.appendChild(row));
    }

    function renderDailyDashboard({ summary, details }) {
        document.getElementById('totalEmployees').textContent = summary.totalEmployees || 0;
        document.getElementById('totalLoggedOut').textContent = summary.loggedOut || 0;
        document.getElementById('totalNotLoggedOut').textContent = summary.notLoggedOut || 0;
        document.getElementById('totalMissingToday').textContent = summary.missing || 0;

        let loggedInUsers = details
            .filter(emp => emp.status !== 'ไม่ได้ลงเวลา')
            .sort((a, b) => a.name.localeCompare(b.name, 'th'));

        const missingUsers = details
            .filter(emp => emp.status === 'ไม่ได้ลงเวลา')
            .sort((a, b) => a.name.localeCompare(b.name, 'th'));

        let onSiteCount = 0;
        let offSiteCount = 0;
        let lateCount = 0;

        loggedInUsers.forEach(emp => {
            if (emp.locationType === 'ปฎิบัติงานในหน่วยบริการ') onSiteCount++;
            else if (emp.locationType === 'ปฎิบัติงานหน่วยบริการ') offSiteCount++;
            if (emp.timeIn && emp.timeIn > LATE_THRESHOLD) lateCount++;
        });

        document.getElementById('totalCheckedIn').textContent = loggedInUsers.length;
        document.getElementById('totalLate').textContent = lateCount;
        document.getElementById('totalOnSite').textContent = onSiteCount;
        document.getElementById('totalOffSite').textContent = offSiteCount;

        const filterValue = document.getElementById('workTypeFilter').value;
        if (filterValue !== 'all') {
            loggedInUsers = loggedInUsers.filter(emp => emp.locationType === filterValue);
        }

        const loggedInTableBody = document.getElementById('dailyAttendanceDetails');
        loggedInTableBody.innerHTML = loggedInUsers.length === 0
            ? '<tr><td colspan="9" class="no-data">ไม่พบข้อมูลตามเงื่อนไขที่เลือก</td></tr>'
            : loggedInUsers.map(entry => {
                const isLate = entry.timeIn && entry.timeIn > LATE_THRESHOLD;
                const checkInStatusText = entry.timeIn ? (isLate ? 'สาย' : 'ปกติ') : '-';
                const checkInStatusBadge = entry.timeIn
                    ? `<span class="status-badge ${isLate ? 'status-late' : 'status-normal'}">${checkInStatusText}</span>`
                    : '-';

                let statusClass = 'status-not-logged-in';
                if (entry.status === 'ออกงานแล้ว') statusClass = 'status-logged-out';
                else if (entry.status === 'ยังไม่ออกงาน') statusClass = 'status-not-logged-out';

                return `
                    <tr>
                        <td data-label="ชื่อ-สกุล">${entry.name}</td>
                        <td data-label="เวลาเข้า">${entry.timeIn || '-'}</td>
                        <td data-label="สถานะเข้างาน">${checkInStatusBadge}</td>
                        <td data-label="รูปเข้า">
                            ${entry.imageIn
                                ? `<img src="${entry.imageIn}" alt="รูปเข้า" style="height:50px;border-radius:8px;cursor:pointer;" onclick="showImageModal('${entry.imageIn}')">`
                                : '-'}
                        </td>
                        <td data-label="เวลาออก">${entry.timeOut || '-'}</td>
                        <td data-label="รูปออก">
                            ${entry.imageOut
                                ? `<img src="${entry.imageOut}" alt="รูปออก" style="height:50px;border-radius:8px;cursor:pointer;" onclick="showImageModal('${entry.imageOut}')">`
                                : '-'}
                        </td>
                        <td data-label="ประเภทงาน">${entry.locationType || 'ไม่ระบุ'}</td>
                        <td data-label="สถานที่">${entry.location || '-'}</td>
                        <td data-label="สถานะ">
                            <span class="status-badge ${statusClass}">${entry.status}</span>
                        </td>
                    </tr>
                `;
            }).join('');

        const missingTableBody = document.getElementById('missingAttendanceDetails');
        missingTableBody.innerHTML = missingUsers.length === 0
            ? '<tr><td colspan="2" class="no-data positive"><i class="fas fa-check-circle"></i> ทุกคนลงเวลาเรียบร้อย</td></tr>'
            : missingUsers.map(entry => `
                <tr>
                    <td data-label="ชื่อ-สกุล">${entry.name}</td>
                    <td data-label="สถานะ"><span class="status-badge status-not-logged-in">${entry.status}</span></td>
                </tr>
            `).join('');

        renderChart(summary);
    }

    function loadIndividualData() {
        const employeeName = document.getElementById('employeeSelect').value;
        const startDate = document.getElementById('startDate').value;
        const endDate = document.getElementById('endDate').value;

        if (!employeeName) {
            if (typeof Swal !== 'undefined') Swal.fire('ข้อมูลไม่ครบ', 'กรุณาเลือกชื่อพนักงาน', 'warning');
            return;
        }
        if (!startDate || !endDate) {
            if (typeof Swal !== 'undefined') Swal.fire('ข้อมูลไม่ครบ', 'กรุณาเลือกช่วงวันที่', 'warning');
            return;
        }

        showLoading(`กำลังค้นหาข้อมูลของ ${employeeName}...`);
        google.script.run
            .withSuccessHandler(results => {
                renderIndividualResults(results, employeeName);
                if (typeof Swal !== 'undefined') Swal.close();
            })
            .withFailureHandler(error => {
                if (typeof Swal !== 'undefined') {
                    Swal.fire('เกิดข้อผิดพลาด', `ไม่สามารถค้นหาข้อมูลได้: ${error.message}`, 'error');
                } else {
                    alert('ไม่สามารถค้นหาข้อมูลได้');
                }
            })
            .getIndividualAttendanceData(employeeName, startDate, endDate);
    }

// === โหลดค่าพิกัดและรัศมีจากชีต 'รัศมี' มาแสดงในฟอร์ม ===
function loadRadiusConfig() {
    showLoading('กำลังโหลดข้อมูลรัศมี...');
    google.script.run
        .withSuccessHandler(data => {
            if (typeof Swal !== 'undefined') Swal.close();

            document.getElementById('radiusCoords').value = data.coords || '';
            document.getElementById('radiusValue').value = data.radius !== '' ? data.radius : '';
        })
        .withFailureHandler(error => {
            if (typeof Swal !== 'undefined') {
                Swal.fire('เกิดข้อผิดพลาด', `ไม่สามารถโหลดข้อมูลรัศมีได้: ${error.message}`, 'error');
            } else {
                alert('ไม่สามารถโหลดข้อมูลรัศมีได้');
            }
        })
        .getLocationConfig();
}

// === บันทึกค่าพิกัดและรัศมีกลับไปยังชีต ===
function saveRadiusConfig() {
    const coords = document.getElementById('radiusCoords').value.trim();
    const radius = document.getElementById('radiusValue').value.trim();

    if (!coords || !radius) {
        if (typeof Swal !== 'undefined') {
            Swal.fire('ข้อมูลไม่ครบ', 'กรุณากรอกพิกัดและรัศมีให้ครบถ้วน', 'warning');
        } else {
            alert('กรุณากรอกพิกัดและรัศมีให้ครบถ้วน');
        }
        return;
    }

    showLoading('กำลังบันทึกข้อมูลรัศมี...');
    google.script.run
        .withSuccessHandler(res => {
            if (typeof Swal !== 'undefined') {
                Swal.fire('สำเร็จ!', res.message, 'success');
            } else {
                alert(res.message);
            }
        })
        .withFailureHandler(error => {
            if (typeof Swal !== 'undefined') {
                Swal.fire('เกิดข้อผิดพลาด', error.message, 'error');
            } else {
                alert(error.message);
            }
        })
        .saveLocationConfig(coords, radius);
}

// === ล้างค่าพิกัดและรัศมี ===
function clearRadiusConfig() {
    if (typeof Swal !== 'undefined') {
        Swal.fire({
            title: 'ยืนยันการล้างค่า',
            text: 'ต้องการล้างค่าพิกัดและรัศมีทั้งหมดหรือไม่?',
            icon: 'warning',
            showCancelButton: true,
            confirmButtonText: 'ใช่, ล้างค่า',
            cancelButtonText: 'ยกเลิก',
            confirmButtonColor: '#ef4444'
        }).then(result => {
            if (result.isConfirmed) {
                showLoading('กำลังล้างค่าพิกัดและรัศมี...');
                google.script.run
                    .withSuccessHandler(res => {
                        if (typeof Swal !== 'undefined') {
                            Swal.fire('สำเร็จ!', res.message, 'success');
                        } else {
                            alert(res.message);
                        }
                        document.getElementById('radiusCoords').value = '';
                        document.getElementById('radiusValue').value = '';
                    })
                    .withFailureHandler(error => {
                        if (typeof Swal !== 'undefined') {
                            Swal.fire('เกิดข้อผิดพลาด', error.message, 'error');
                        } else {
                            alert(error.message);
                        }
                    })
                    .clearLocationConfig();
            }
        });
    } else {
        if (!confirm('ต้องการล้างค่าพิกัดและรัศมีทั้งหมดหรือไม่?')) return;
        google.script.run.clearLocationConfig();
        document.getElementById('radiusCoords').value = '';
        document.getElementById('radiusValue').value = '';
        alert('ล้างค่าพิกัดและรัศมีเรียบร้อยแล้ว');
    }
}
