    function updateDateTime() {
      const now = new Date();
      const dateOptions = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
      const timeOptions = { hour: '2-digit', minute: '2-digit', second: '2-digit' };
      const formattedDate = now.toLocaleDateString('th-TH', dateOptions);
      const formattedTime = now.toLocaleTimeString('th-TH', timeOptions);
      currentTimeDisplay.textContent = formattedTime;
      currentDateDisplay.textContent = formattedDate;
      infoDateDisplay.textContent = formattedDate;
    }

    function getGeoLocation() {
      if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(
          pos => {
            latitude = pos.coords.latitude;
            longitude = pos.coords.longitude;
            document.getElementById('lat').textContent = latitude.toFixed(6);
            document.getElementById('lng').textContent = longitude.toFixed(6);
          },
          err => {
            document.getElementById('lat').textContent = 'ไม่สามารถระบุ';
            document.getElementById('lng').textContent = 'ตำแหน่งได้';
            Swal.fire({
              icon: 'warning',
              title: 'ไม่พบตำแหน่ง',
              text: 'ไม่สามารถเข้าถึงข้อมูล GPS ได้ โปรดอนุญาตการเข้าถึงตำแหน่ง'
            });
          },
          { enableHighAccuracy: true, timeout: 5000, maximumAge: 0 }
        );
      }
    }

    function verifyAdminCode() {
      Swal.fire({
        title: 'กรอกรหัสผู้ดูแลระบบ',
        input: 'password',
        inputPlaceholder: 'พิมพ์รหัสผ่านที่นี่',
        showCancelButton: true,
        confirmButtonText: 'ยืนยัน',
        cancelButtonText: 'ยกเลิก',
        preConfirm: (inputCode) => {
          if (!inputCode) {
            Swal.showValidationMessage('กรุณากรอกรหัสผ่าน');
            return false;
          }
          showSpinner();
          return new Promise((resolve) => {
            if (typeof google !== 'undefined' && google.script && google.script.run) {
              google.script.run
                .withSuccessHandler(storedCode => {
                  hideSpinner();
                  if (inputCode === storedCode) {
                    resolve(true);
                  } else {
                    Swal.showValidationMessage('รหัสผ่านไม่ถูกต้อง');
                    resolve(false);
                  }
                })
                .withFailureHandler(err => {
                  hideSpinner();
                  Swal.showValidationMessage('เกิดข้อผิดพลาด: ' + err.message);
                  resolve(false);
                })
                .getAdminCode();
            } else {
              hideSpinner();
              if (inputCode === '1') {
                resolve(true);
              } else {
                Swal.showValidationMessage('รหัสผ่านไม่ถูกต้อง (โหมดจำลอง)');
                resolve(false);
              }
            }
          });
        }
      }).then(result => {
        if (result.isConfirmed) {
          toggleContent();
        }
      });
    }

    /* ===== Name Picker Logic ===== */
    const debounce = (fn, ms = 200) => {
      let t;
      return (...a) => {
        clearTimeout(t);
        t = setTimeout(() => fn(...a), ms);
      };
    };
    const norm = s => (s || '').toString().toLowerCase().trim();

    function highlight(text, query) {
      if (!query) return text;
      const q = query.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
      const re = new RegExp(`(${q})`, 'ig');
      return text.replace(re, '<mark>$1</mark>');
    }

    function getFav() {
      try { return JSON.parse(localStorage.getItem(LS_FAV) || '[]'); }
      catch (e) { return []; }
    }
    function setFav(arr) {
      localStorage.setItem(LS_FAV, JSON.stringify(arr));
    }
    function getRecent() {
      try { return JSON.parse(localStorage.getItem(LS_RECENT) || '[]'); }
      catch (e) { return []; }
    }

    // ปุ่มล้างรายชื่อล่าสุด
    function clearRecent() {
      localStorage.removeItem(LS_RECENT);
      renderChips();
    }

    function pushRecent(name) {
      let r = getRecent().filter(x => x !== name);
      r.unshift(name);
      if (r.length > MAX_RECENT) r = r.slice(0, MAX_RECENT);
      localStorage.setItem(LS_RECENT, JSON.stringify(r));
    }

    function renderChips() {
      const fav = getFav();
      const rec = getRecent();
      quickChips.innerHTML = '';

      if (fav.length) {
        const tag = document.createElement('span');
        tag.className = 'text-muted me-2';
        tag.textContent = 'รายการโปรด:';
        quickChips.appendChild(tag);
        fav.forEach(n => addChip(n));
      }
      if (rec.length) {
        const sep = document.createElement('span');
        sep.className = 'text-muted ms-3 me-2';
        sep.textContent = 'รายชื่อล่าสุดที่ใช้:';
        quickChips.appendChild(sep);

        // ปุ่มล้าง
const clearBtn = document.createElement('button');
clearBtn.type = 'button';
clearBtn.className = 'btn btn-link btn-sm p-0 me-2';
clearBtn.innerHTML = '<i class="fa fa-trash"></i>'; 
clearBtn.onclick = clearRecent;
quickChips.appendChild(clearBtn);
rec.forEach(n => addChip(n));
      }

      function addChip(name) {
        const b = document.createElement('button');
        b.type = 'button';
        b.className = 'chip';
        b.textContent = name;
        b.onclick = () => selectEmployee(name);

    function selectEmployee(name) {
      nameSelect.value = name;
      nameSearchInput.value = name;
      if (name) pushRecent(name);
      renderChips();
      hideDropdown();
    }

    function togglePin(name, btn) {
      const fav = getFav();
      const idx = fav.indexOf(name);
      if (idx > -1) {
        fav.splice(idx, 1);
        btn.classList.remove('pinned');
      } else {
        fav.push(name);
        btn.classList.add('pinned');
      }
      setFav(fav);
      renderChips();
    }

    function renderDropdown(list, query) {
      nameDropdown.innerHTML = '';
      if (!list.length) {
        const empty = document.createElement('div');
        empty.className = 'dropdown-item text-muted';
        empty.textContent = 'ไม่พบรายชื่อที่ตรง';
        nameDropdown.appendChild(empty);
      } else {
        const fav = getFav();
        list.forEach((n, i) => {
          const item = document.createElement('div');
          item.className = 'item';
          item.setAttribute('role', 'button');
          item.innerHTML = `
            <i class="fa-regular fa-user"></i>
            <div class="flex-1">
              <div class="name">${highlight(n, query)}</div>
            </div>
            <button class="pin-btn ${fav.includes(n) ? 'pinned' : ''}" title="ปักหมุด">
              <i class="fa-solid fa-star"></i>
            </button>
          `;
          item.addEventListener('click', (e) => {
            if (e.target.closest('.pin-btn')) return;
            selectEmployee(n);
          });
          item.querySelector('.pin-btn').addEventListener('click', (e) => {
            e.stopPropagation();
            togglePin(n, e.currentTarget);
          });
          nameDropdown.appendChild(item);
        });
      }
      showDropdown();
      activeIndex = -1;
      updateActiveItem();
    }

    function showDropdown() {
      nameDropdown.style.display = 'block';
    }
    function hideDropdown() {
      nameDropdown.style.display = 'none';
      activeIndex = -1;
      updateActiveItem();
    }

    function updateActiveItem() {
      [...nameDropdown.querySelectorAll('.item')].forEach((el, idx) => {
        el.classList.toggle('active', idx === activeIndex);
      });
    }

    const filterNames = debounce((q) => {
      const qq = norm(q);
      if (!qq) {
        filtered = EMPLOYEES.slice(0, 50);
      } else {
        filtered = EMPLOYEES.filter(n => norm(n).includes(qq)).slice(0, 100);
      }
      renderDropdown(filtered, q);
    }, 200);

    nameSearchInput.addEventListener('input', e => filterNames(e.target.value));
    nameSearchInput.addEventListener('focus', () => {
      filterNames(nameSearchInput.value);
      showDropdown();
    });

    nameSearchInput.addEventListener('keydown', (e) => {
      const items = nameDropdown.querySelectorAll('.item');
      if (e.key === 'ArrowDown') {
        e.preventDefault();
        if (items.length) {
          activeIndex = Math.min(items.length - 1, activeIndex + 1);
          updateActiveItem();
          items[activeIndex]?.scrollIntoView({ block: 'nearest' });
        }
      } else if (e.key === 'ArrowUp') {
        e.preventDefault();
        if (items.length) {
          activeIndex = Math.max(0, activeIndex - 1);
          updateActiveItem();
          items[activeIndex]?.scrollIntoView({ block: 'nearest' });
        }
      } else if (e.key === 'Enter') {
        if (activeIndex > -1 && filtered[activeIndex]) {
          e.preventDefault();
          selectEmployee(filtered[activeIndex]);
        }
      } else if (e.key === 'Escape') {
        hideDropdown();
      }
    });

    document.addEventListener('click', (e) => {
      if (!e.target.closest('#nameDropdown') && !e.target.closest('#nameSearch')) {
        hideDropdown();
      }
    });
